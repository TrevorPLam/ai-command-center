    asset_id: "[ASSET_ID]"   # for CREATE/OVERWRITE/BOOTSTRAP
    operation: CREATE
    expected_sha256: "[hash-or-PENDING]"

    # For all operations, if expected_sha256 is PENDING, it will be set by a bootstrap step.

    bootstrap:

        If enabled and expected_sha256 is "PENDING", the agent will compute the

      on_first_run:
        if_expected_sha256_is: "PENDING"

          Agent computes SHA256 of staged file, updates expected_sha256 in this

## SECTION 8 — PROCEDURE

```yaml
PROCEDURE:
  1:
    action: PRE_FLIGHT
    description: |
      1) Verify SAME_MODULE and CROSS_MODULE dependencies.
      2) Read MASTER-CONTRACTS.md and confirm CONTRACTS.source_version matches.
      3) Validate that no FILE_MANIFEST entry has both bootstrap.enabled and variable_output.enabled.
    on_fail: STOP

  2:
    action: IDEMPOTENCY_SCAN
    description: |
      For each file in FILE_MANIFEST with operation in
      [CREATE, OVERWRITE, PATCH, BOOTSTRAP, VARIABLE_OUTPUT]:
        - If file exists, compute hash using available tools.
        - If expected_sha256 is NOT "PENDING" and hash matches expected_sha256, mark file as SKIP.
        - If expected_sha256 is "PENDING", do NOT skip (bootstrap will set hash).
        - Log which files will be written.
    on_fail: CONTINUE

  2.5:
    action: ACQUIRE_LOCKS
    description: |
      For each FILE_MANIFEST entry, determine lock_name from FILE_LOCKING.scope
      and create lock directory at .agent/locks/[lock_name] using atomic mkdir.
      If mkdir fails because directory exists, retry with exponential backoff (1s,2s,4s) up to 3 attempts.
      If all retries fail, STOP.
    on_fail: STOP

  3:
    action: PREPARE_STAGING_AREA
    description: |
      Create a clean staging directory at ".agent/staging/[TASK-ID]/".
      All writes for this task go into staging first.
    on_fail: STOP

  4:
    action: DISCOVER         # include only if any DISCOVER entries exist
    description: |
      Perform bounded investigation to populate DISCOVER artifacts.
      Use only commands listed in discover.allowed_side_effects.
      MUST NOT modify tracked source files.
    on_fail: STOP

  5:
    action: WRITE_STAGED_FILES
    description: |
      For each non-SKIP file in FILE_MANIFEST:
        - Render content from its Code Block (if applicable).
        - Write to ".agent/staging/[TASK-ID]/[path]".
        - Enforce LF endings, UTF-8 no BOM.
        - Run GUARDS that operate on single files.
        - If bootstrap.enabled and expected_sha256 is PENDING:
          compute SHA256 of the freshly written file, then update the task spec file:
          set expected_sha256 to that hash and bootstrap.enabled to false.
          (This edit is done directly on the task file; the agent must not re-read
          the task file after this step.)
    on_fail: |
      STOP — keep staging for debugging; do NOT commit staged files.

  6:
    action: COMMIT_STAGED_FILES
    description: |
      For each staged file:
        - Move from staging to real workspace (atomic rename).
        - For PATCH: verify patch_guard hash (file must exist), apply patch, and then
          verify the new hash matches expected_sha256 (post-patch).
    on_fail: |
      STOP — workspace may be partially updated; record committed files.

  7:
    action: RUN_ADDITIONAL_COMMANDS
    allowlist: ["pnpm build", "pnpm test", "pnpm lint", "tsc --noEmit"]
    dry_run_default: true
    commands:
      - cmd: "pnpm build"
        dry_run: false
        timeout: 600
        success_signal: "Build completed"
        on_timeout: "STOP — build timed out"
        on_fail: STOP

  8:
    action: RUN_VERIFICATION
    description: "Execute all VERIFICATION commands from SECTION 9."
    on_fail: |
      Apply relevant SELF_HEALING_RECIPES via classifier; retry once per command.
      If still failing, STOP.

  9:
    action: WRITE_COMPLETION_ARTIFACTS
    description: "Execute ON_COMPLETE steps in SECTION 10."
    on_fail: STOP

  10:
    action: RELEASE_LOCKS
    description: "Delete all lock directories created in .agent/locks/ for this task."
    on_fail: "WARN — lock release failed; requires manual cleanup."
```
