## 📦 AGENT PREAMBLE)
══════════════════════════════════════════════════════════════════════════════
- **This specification is incomplete until the bootstrapping task (FND‑BOOT) runs.**
  The agent MUST execute FND‑BOOT before any other task. FND‑BOOT will:
    a) Compute the SHA‑256 hash of every code block embedded in this specification,
       treating each block as a file to be created.
    b) Generate a `pnpm‑lock.yaml` from the exact dependency versions listed in
       `sec‑FND‑002` and compute its SHA‑256 hash.
    c) Generate any external generated files (e.g., shadcn Button) and capture their
       exact content and hashes.
    d) Insert all computed hashes back into this specification file, replacing the
       `H(<asset‑id>)` place‑holders.
    e) Re‑verify that the specification is now internally consistent (no remaining
       place‑holders, no missing code blocks).
  After FND‑BOOT, the specification is fully deterministic and can be re‑executed
  from scratch using only this document.

- **Execution model:** The agent ensures that every file in the target workspace
  matches the SHA‑256 hash recorded in this specification. If a file exists and its
  hash matches, the task is skipped. Otherwise, the agent creates or overwrites the
  file with the exact content from the associated code block. This is called
  **idempotent reconciliation**.

- **Resource locking:** Each task declares the files it reads and writes. The
  scheduler MUST NOT run two tasks concurrently if they write to the same file, or
  if one writes a file that another reads (read‑after‑write dependency). The
  dependency graph in the MODULE REGISTRY overrules parallel groups when there
  is a conflict.

- **Self‑healing & fix‑it:** Guard failures of severity [B] (BUG) have associated
  `fixit` recipes given in the task block. The agent applies the recipe up to 2
  times and re‑verifies. If it fails, the task stops.
  A global **Self‑Healing Recipe Catalogue** is provided in Appendix A.

- **Cryptographic attestation:** After every task, the agent signs an attestation
  containing:
    - Task ID and its subtree of completed sub‑tasks
    - SHA‑256 of every file modified
    - Merkle root of the entire workspace (excluding `node_modules` and `.agent/`)
    - Timestamp and execution log
  Attestations are stored in `.agent/attestations/` and are chained.

- **Parallel execution:** Tasks in the same `parallel_group` may run concurrently
  subject to resource locking. Max concurrency = 2.

- **Resume logic:** From the CHECKPOINT BLOCK, if `resume_task` is set, continue
  from that task; otherwise from `next_task`. The last completed task’s workspace
  snapshot (stored in `.agent/checkpoints/`) is used to restore state if a failure
  occurs.

- **Windsurf‑specific hints:** 
  - After changing `package.json`, always run `pnpm install`.
  - If a `tsc --noEmit` fails, ensure that all `@types` packages listed in the
    dependencies are installed.
  - Use `pnpm dev` only to quickly verify the server starts; then kill it before
    proceeding to tests.
  - All code blocks must be saved with LF line endings (not CRLF) and UTF‑8
    without BOM. Emojis must be preserved exactly.

- **Before You Begin (for the human user):**
  1. Ensure Node.js ≥20.11.0 is installed (`node -v`)
  2. Install pnpm 9 (`npm install -g pnpm`)
  3. Create a new empty folder and open it in Windsurf (or your assistant)
  4. Paste this entire document into the chat
  5. Tell the assistant: *“Execute every task in order, starting with FND‑BOOT.”*

  **What you’ll end up with:** A running, dark‑themed Personal AI Command Center 
  with a glass sidebar, command palette, voice orb, status bar, and offline‑first 
  database—ready to be extended with AI modules.

interface SanitizedHTMLProps {
  html: string;
  allowlist?: DOMPurify.Config['ALLOWED_TAGS'];
  className?: string;
}

const DEFAULT_ALLOWLIST = {
  ALLOWED_TAGS: [
    'p','h1','h2','h3','h4','h5','h6','ul','ol','li','strong','em','a',
    'img','blockquote','pre','code','br','hr','table','thead','tbody','tr','th','td'
  ],
  ALLOWED_ATTR: ['href','src','alt','title','target','rel','class'],
};

export function SanitizedHTML({ html, allowlist = DEFAULT_ALLOWLIST, className }: SanitizedHTMLProps) {
  const cleanHtml = DOMPurify.sanitize(html, {
    ALLOWED_TAGS: allowlist.ALLOWED_TAGS,
    ALLOWED_ATTR: allowlist.ALLOWED_ATTR,
    FORBID_TAGS: ['script','iframe','object','embed','form','input'],
    FORBID_ATTR: ['onclick','onload','onerror','onmouseover'],
  });

  const processedHtml = cleanHtml.replace(
    /<a\s+href=(["'])(https?:\/\/[^"']+)\1/gi,
    '<a href="$1$2$1" target="_blank" rel="noopener noreferrer"'
  );

  return <div className={className}>{parse(processedHtml)}</div>;
}

import { useEffect, useState } from 'react';

export function useStorageQuota() {
  const [quota, setQuota] = useState<{
    usage: number; quota: number; percentage: number; isNearLimit: boolean
  } | null>(null);
  const [isLoading, setIsLoading] = useState(true);

  useEffect(() => {
    const estimate = async () => {
      try {
        const est = await navigator.storage.estimate();
        const usage = est.usage || 0;
        const quotaLimit = est.quota || 0;
        const percentage = quotaLimit > 0 ? (usage / quotaLimit) * 100 : 0;
        setQuota({
          usage,
          quota: quotaLimit,
          percentage,
          isNearLimit: percentage >= 80,
        });
      } catch (error) {
        console.warn('Storage estimate failed:', error);
      } finally {
        setIsLoading(false);
      }
    };
    estimate();
    const interval = setInterval(estimate, 5 * 60 * 1000);
    return () => clearInterval(interval);
  }, []);

  const formatBytes = (bytes: number) => {
    const units = ['B','KB','MB','GB'];
    let size = bytes, idx = 0;
    while (size >= 1024 && idx < units.length-1) { size /= 1024; idx++; }
    return `${size.toFixed(1)} ${units[idx]}`;
  };

  return {
    quota,
    isLoading,
    formatBytes,
    shouldShowWarning: quota?.isNearLimit ?? false,
    warningMessage: quota
      ? `Storage usage at ${quota.percentage.toFixed(1)}% (${formatBytes(quota.usage)} / ${formatBytes(quota.quota)})`
      : undefined,
  };
}

import { useMutation, useQueryClient, type QueryKey } from '@tanstack/react-query';

export function useOptimisticMutation<TData, TError, TVariables>({
  mutationFn,
  queryKey,
  optimisticUpdate,
  onSettled,
}: {
  mutationFn: (vars: TVariables) => Promise<TData>;
  queryKey: QueryKey;
  optimisticUpdate: (old: TData | undefined, vars: TVariables) => TData;
  onSettled?: () => void;
}) {
  const queryClient = useQueryClient();
  return useMutation({
    mutationFn,
    onMutate: async (vars) => {
      await queryClient.cancelQueries({ queryKey });
      const previous = queryClient.getQueryData<TData>(queryKey);
      queryClient.setQueryData<TData>(queryKey, (old) => optimisticUpdate(old, vars));
      return { previous };
    },
    onError: (_err, _vars, context) => {
      if (context?.previous !== undefined) {
        queryClient.setQueryData(queryKey, context.previous);
      }
    },
    onSettled: () => {
      queryClient.invalidateQueries({ queryKey });
      onSettled?.();
    },
  });
}
```

import { useState, useEffect, useRef } from 'react';
import { motion, AnimatePresence, useReducedMotion } from 'motion/react';
import { createPortal } from 'react-dom';
import { useIntentHandler } from '@/hooks/useIntentHandler';
import { useFocusTriggerRef, useUIActions } from '@/stores';

export default function CommandPalette() {
  const [open, setOpen] = useState(false);
  const [query, setQuery] = useState('');
  const inputRef = useRef<HTMLInputElement>(null);
  const { intent, parseInput, clearIntent } = useIntentHandler();
  const focusTriggerRef = useFocusTriggerRef();
  const { setFocusTrigger } = useUIActions();
  const prefersReduced = useReducedMotion();

  const transition = prefersReduced ? { duration: 0 } : { type: 'spring', stiffness: 300, damping: 30 };

  useEffect(() => {
    const handler = (e: KeyboardEvent) => {
      if ((e.metaKey || e.ctrlKey) && e.key === 'k') {
        e.preventDefault();
        setOpen((prev) => !prev);
      }
      if (e.key === 'Escape' && open) {
        setOpen(false);
        focusTriggerRef?.focus();
      }

    if (open) {

      setFocusTrigger(document.activeElement as HTMLElement);

    setQuery(e.target.value);
    if (e.target.value) parseInput(e.target.value);
    else clearIntent();
  };

        <motion.div

            role="combobox"
            aria-expanded={open}
            aria-haspopup="listbox"
          >
            <input
              ref={inputRef}
              type="text"
              value={query}
              onChange={handleInput}
              placeholder="Type a command..."
              className="w-full bg-transparent border-b border-muted px-4 py-3 text-sm outline-none"
            />

              <div className="px-4 py-2 text-xs text-muted-foreground">

import { useState, useEffect, useCallback } from 'react';

type VoiceState = 'idle' | 'listening' | 'processing' | 'preview' | 'execute';

export function useWebSpeech() {
  const [state, setState] = useState<VoiceState>('idle');
  const [transcript, setTranscript] = useState('');
  const [isSupported, setIsSupported] = useState(true);

  useEffect(() => {
    if (!('SpeechRecognition' in window || 'webkitSpeechRecognition' in window)) {
      setIsSupported(false);
    }
  }, []);

  const startListening = useCallback(() => {
    if (!isSupported) return;
    setState('listening');
    const timer = setTimeout(() => {
      setTranscript('This is a test command');
      setState('processing');
      setTimeout(() => setState('preview'), 1000);
    }, 2000);
    return () => clearTimeout(timer);
  }, [isSupported]);

  const confirm = useCallback(() => {
    setState('execute');
    setTimeout(() => setState('idle'), 1000);
  }, []);

  const cancel = useCallback(() => {
    setState('idle');
    setTranscript('');
  }, []);

  const speak = useCallback((text: string) => {
    if ('speechSynthesis' in window) {
      const utterance = new SpeechSynthesisUtterance(text);
      window.speechSynthesis.speak(utterance);
    }
  }, []);

  return { state, transcript, isSupported, startListening, confirm, cancel, speak };
}
```
